编写一个递归实现生成多级子目录

 public List<PmsCategory> createChildren(PmsCategory pm,List<PmsCategory> pms){
          List<PmsCategory>pc=pms.stream()
           //过滤出所有父Id等于pmId的对象，即子目录
                          .filter(item->{
                           item.getParentId==pm.getParentId;
                  })
                          .map(item->{
                               //递归实现子目录下子目录
                                 item.setList(createChildren(item,pms));
                                 return item;
                   })
                   //转为List集合
                         .collect(Collectors.toList());
            return pc;
}
